# 1жҢҮй’ҲеҒҡеҮҪж•°зҡ„еҪўеҸӮ

жҢҮй’ҲеҸҳйҮҸеҸҜд»ҘеҒҡеҮҪж•°зҡ„еҪўеҸӮпјҢеҜ№еә”зҡ„е®һеҸӮз”Ёең°еқҖжҲ–жҢҮй’ҲеҸҳйҮҸгҖӮеҮҪж•°и°ғз”Ёж—¶дј е…Ҙзҡ„жҳҜж•°жҚ®зҡ„ең°еқҖпјҢеӣ жӯӨеҪўеҸӮе’Ңе®һеҸӮжҢҮеҗ‘дё»и°ғеҮҪж•°дёӯзҡ„еҗҢдёҖзүҮеҶ…еӯҳз©әй—ҙпјҢеҮҪж•°ж”№еҸҳеҪўеҸӮжҢҮеҗ‘зҡ„еҸҳйҮҸзҡ„еҖјпјҢеҲҷеҜ№еә”е®һеҸӮжҢҮеҗ‘зҡ„еҖјд№ҹе°Ҷж”№еҸҳгҖӮеӣ жӯӨпјҢе°Ҫз®Ўиў«и°ғеҮҪж•°дёҚиғҪж”№еҸҳе®һеҸӮжҢҮй’Ҳзҡ„еҖјпјҢдҪҶеҸҜж”№еҸҳе®һеҸӮжҢҮй’ҲжүҖжҢҮеҸҳйҮҸзҡ„еҖјгҖӮ

# 2.иҝ”еӣһжҢҮй’ҲеҖјзҡ„еҮҪж•°

еҮҪж•°зҡ„иҝ”еӣһеҖјеҸҜд»ҘжҳҜжҢҮй’ҲпјҢиҝ”еӣһеҖјзұ»еһӢжҳҜжҢҮй’Ҳзҡ„еҮҪж•°д№ҹиў«з§°дёәжҢҮй’ҲеҮҪж•°гҖӮеӣ жң¬еҮҪж•°жү§иЎҢе®ҢжҜ•иҝ”еӣһдё»и°ғеҮҪж•°еҗҺпјҢеҮҪж•°дёӯзҡ„еұҖйғЁеҸҳйҮҸе°Ҷиў«е…ЁйғЁйҮҠж”ҫпјҢеӣ жӯӨжҢҮй’ҲеҮҪж•°зҡ„иҝ”еӣһеҖјдёҚиғҪжҳҜжң¬еҮҪж•°дёӯзҡ„еұҖйғЁеҸҳйҮҸзҡ„ең°еқҖгҖӮ

# 3.жҢҮеҗ‘еҮҪж•°зҡ„жҢҮй’Ҳ

зЁӢеәҸиҝҗиЎҢж—¶пјҢдёҖдёӘеҮҪж•°еҢ…еҗ«зҡ„жҢҮд»ӨеәҸеҲ—иҰҒеҚ жҚ®дёҖж®өеҶ…еӯҳз©әй—ҙпјҢиҝҷеқ—еҶ…еӯҳз©әй—ҙзҡ„йҰ–ең°еқҖз§°дёәеҮҪж•°зҡ„е…ҘеҸЈең°еқҖпјҢзј–иҜ‘зі»з»ҹз”ЁеҮҪж•°еҗҚд»ЈиЎЁиҝҷдёҖең°еқҖгҖӮеҮҪж•°зҡ„е…ҘеҸЈең°еқҖдёәеҮҪж•°зҡ„жҢҮй’ҲпјҢз®Җз§°еҮҪж•°жҢҮй’ҲгҖӮеҮҪж•°еҗҚжҳҜеҮҪж•°зҡ„жҢҮй’ҲеёёйҮҸпјҢеҸҜд»Ҙе®ҡд№үжҢҮй’ҲеҸҳйҮҸпјҢз”ЁжқҘеӯҳж”ҫеҮҪж•°зҡ„е…ҘеҸЈең°еқҖпјҢиҝҷз§Қеӯҳж”ҫеҮҪж•°е…ҘеҸЈең°еқҖзҡ„еҸҳйҮҸз§°дёәжҢҮеҗ‘еҮҪж•°зҡ„жҢҮй’ҲеҸҳйҮҸпјҢз®Җз§°еҮҪж•°жҢҮй’ҲеҸҳйҮҸгҖӮ

з”ЁеҮҪж•°еҗҚи°ғз”ЁеҮҪж•°з§°дёәеҮҪж•°зҡ„зӣҙжҺҘи°ғз”ЁпјҢз”ЁеҮҪж•°жҢҮй’ҲеҸҳйҮҸи°ғз”ЁеҮҪж•°з§°дёәеҮҪж•°зҡ„й—ҙжҺҘи°ғз”ЁгҖӮ

дёҖдёӘеҮҪж•°жҢҮй’ҲеҸҳйҮҸеҸӘиғҪжҢҮеҗ‘жҹҗзұ»еҮҪж•°дёӯзҡ„дёҖдёӘпјҢиҰҒжұӮиҜҘеҮҪж•°зҡ„еҪўеҸӮдёӘж•°\зұ»еһӢ\йЎәеәҸзӣёеҗҢдё”иҝ”еӣһеҖјзұ»еһӢд№ҹзӣёеҗҢпјҢеҸӘжҳҜеҮҪж•°еҗҚдёҚеҗҢгҖӮ

еҮҪж•°жҢҮй’ҲдёҺжҷ®йҖҡзұ»еһӢеҸҳйҮҸжҢҮй’Ҳжң¬иҙЁзӣёеҗҢпјҢйғҪжҳҜеҶ…еӯҳең°еқҖпјҢеҢәеҲ«жҳҜжҷ®йҖҡеҸҳйҮҸзҡ„жҢҮй’ҲжҢҮеҗ‘еҶ…еӯҳдёӯзҡ„ж•°жҚ®еҢәпјҢеҮҪж•°жҢҮй’ҲжҢҮеҗ‘еҶ…еӯҳдёӯзҡ„еҮҪж•°жҢҮд»ӨеҢәгҖӮ

# 4.д»Јз ҒзӨәдҫӢ

#include <iostream>

using namespace std;

void swap(int *a, int *b); 
int *addArray(int *array, size_t size);
int *ptrParamAddArray(int *array, size_t size, int *(*p)(int *, size_t));

int main(int argc, char **argv)
{
    // жҢҮй’ҲеҸҳйҮҸеҒҡеҮҪж•°еҪўеҸӮ
    int x = 1, y = 2;
    cout << "before swap: " << "(x,y) = " << x << "," << y << endl;
    swap(&x, &y);
    cout << "after swap: " << "(x,y) = " << x << "," << y << endl;
    
    // жҢҮй’ҲеҸҳйҮҸеҒҡеҮҪж•°еҪўеҸӮпјҢиҝ”еӣһеҖјжҳҜжҢҮй’Ҳзұ»еһӢзҡ„еҮҪж•°
    int a[2] = {2, 3};
    cout << "before add: " << "(a[0], a[1])пјҡ " << a[0] << "," << a[1] << endl;
    int *p = addArray(a, 2);
    cout << "after add: " << "(a[0], a[1])пјҡ " << p[0] << "," << p[1] << endl;

    // е®ҡд№үжҢҮеҗ‘еҮҪж•°зҡ„жҢҮй’ҲеҸҳйҮҸ
    void (*fp)(int*, int*);
    fp = swap;
    cout << "before pswap: " << "(x,y) = " << x << "," << y << endl;
    fp(&x, &y);
    cout << "after pswap: " << "(x,y) = " << x << "," << y << endl;

    // е®ҡд№үжҢҮеҗ‘иҝ”еӣһеҖјжҳҜжҢҮй’Ҳзұ»еһӢеҮҪж•°зҡ„жҢҮй’ҲеҸҳйҮҸ
    int *(*fpa)(int *, size_t);
    fpa = addArray;
    cout << "before fpa add: " << "(a[0], a[1])пјҡ " << a[0] << "," << a[1] << endl;
    p = fpa(a, 2);
    cout << "after fpa add: " << "(a[0], a[1])пјҡ " << p[0] << "," << p[1] << endl;

    // еҮҪж•°жҢҮй’ҲдҪңдёәеҮҪж•°зҡ„еҪўеҸӮ
    cout << "before pfpa add: " << "(a[0], a[1])пјҡ " << a[0] << "," << a[1] << endl;
    p = ptrParamAddArray(a, 2, fpa);
    cout << "after pfpa add: " << "(a[0], a[1])пјҡ " << p[0] << "," << p[1] << endl;

    return 0;

}

// жҢҮй’ҲеҒҡеҮҪж•°еҸӮж•°
void swap(int *a, int *b)
{
    int t = *a;
    *a = *b;
    *b = t;
}

// иҝ”еӣһжҢҮй’Ҳзҡ„еҮҪж•°
int *addArray(int *array, size_t size)
{
    for(size_t i = 0; i < size; i++)
    {
        array[i] += i;
    }
    return array;
    // // return pointer of local variable[error] 
    // int p[2] = {4, 5};
    // int *r = p;
    // return r;
}

// еҮҪж•°жҢҮй’ҲеҒҡеҮҪж•°еҪўеҸӮзҡ„еҮҪж•°
int *ptrParamAddArray(int *array, size_t size, int *(*p)(int *, size_t))
{
    int *tmp = p(array, size);
    array = tmp;
    return array;
}
